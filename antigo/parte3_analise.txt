    "def resolver_cvrp(n_veiculos, coords, dem, cap, nomes_pontos):\n",
    "    n = len(coords)\n",
    "    cli = list(range(1, n))\n",
    "    pts = list(range(n))\n",
    "    veics = list(range(n_veiculos))\n",
    "    d = [[distancia(coords[i], coords[j]) for j in range(n)] for i in range(n)]\n",
    "\n",
    "    m = LpProblem(f'CVRP_{n_veiculos}v', LpMinimize)\n",
    "    xv = LpVariable.dicts('x', (pts, pts, veics), cat='Binary')\n",
    "    uv = LpVariable.dicts('u', (cli, veics), lowBound=1, upBound=len(cli), cat='Continuous')\n",
    "\n",
    "    m += lpSum(d[i][j] * xv[i][j][k] for i in pts for j in pts for k in veics if i != j)\n",
    "\n",
    "    for j in cli:\n",
    "        m += lpSum(xv[i][j][k] for i in pts for k in veics if i != j) == 1\n",
    "    for k in veics:\n",
    "        m += lpSum(xv[0][j][k] for j in cli) <= 1\n",
    "        m += lpSum(xv[j][0][k] for j in cli) <= 1\n",
    "    for k in veics:\n",
    "        for j in cli:\n",
    "            m += lpSum(xv[i][j][k] for i in pts if i != j) == lpSum(xv[j][i][k] for i in pts if i != j)\n",
    "    for k in veics:\n",
    "        m += lpSum(dem[j] * lpSum(xv[i][j][k] for i in pts if i != j) for j in cli) <= cap\n",
    "    for k in veics:\n",
    "        for i in cli:\n",
    "            for j in cli:\n",
    "                if i != j:\n",
    "                    m += uv[i][k] - uv[j][k] + len(cli) * xv[i][j][k] <= len(cli) - 1\n",
    "\n",
    "    m.solve(PULP_CBC_CMD(msg=0))\n",
    "    rotas_r = extrair_rotas(xv, pts, cli, veics)\n",
    "    return m, rotas_r"
   ]
