    "clientes = list(range(1, N))\n",
    "pontos = list(range(N))\n",
    "veiculos = list(range(N_VEICULOS))\n",
    "\n",
    "modelo = LpProblem('CVRP', LpMinimize)\n",
    "\n",
    "x = LpVariable.dicts('x', (pontos, pontos, veiculos), cat='Binary')\n",
    "u = LpVariable.dicts('u', (clientes, veiculos), lowBound=1, upBound=N_CLIENTES, cat='Continuous')\n",
    "\n",
    "# FUNÇÃO OBJETIVO\n",
    "modelo += lpSum(dist[i][j] * x[i][j][k] for i in pontos for j in pontos for k in veiculos if i != j)\n",
    "\n",
    "# RESTRIÇÃO 1: cada cliente visitado exatamente 1 vez\n",
    "for j in clientes:\n",
    "    modelo += lpSum(x[i][j][k] for i in pontos for k in veiculos if i != j) == 1\n",
    "\n",
    "# RESTRIÇÃO 2: cada veículo sai do CD no máximo 1 vez\n",
    "for k in veiculos:\n",
    "    modelo += lpSum(x[0][j][k] for j in clientes) <= 1\n",
    "\n",
    "# RESTRIÇÃO 3: cada veículo retorna ao CD no máximo 1 vez\n",
    "for k in veiculos:\n",
    "    modelo += lpSum(x[j][0][k] for j in clientes) <= 1\n",
    "\n",
    "# RESTRIÇÃO 4: conservação de fluxo\n",
    "for k in veiculos:\n",
    "    for j in clientes:\n",
    "        modelo += lpSum(x[i][j][k] for i in pontos if i != j) == lpSum(x[j][i][k] for i in pontos if i != j)\n",
    "\n",
    "# RESTRIÇÃO 5: capacidade do veículo\n",
    "for k in veiculos:\n",
    "    modelo += lpSum(demandas[j] * lpSum(x[i][j][k] for i in pontos if i != j) for j in clientes) <= CAPACIDADE\n",
    "\n",
    "# RESTRIÇÃO 6: eliminação de sub-rotas (MTZ)\n",
    "for k in veiculos:\n",
    "    for i in clientes:\n",
    "        for j in clientes:\n",
    "            if i != j:\n",
    "                modelo += u[i][k] - u[j][k] + N_CLIENTES * x[i][j][k] <= N_CLIENTES - 1"

    "cores = ['green', 'purple']\n",
    "\n",
    "def extrair_rotas(x, pontos, clientes, veiculos):\n",
    "    rotas = {}\n",
    "    for k in veiculos:\n",
    "        rota = [0]\n",
    "        atual = 0\n",
    "        while True:\n",
    "            proximo = None\n",
    "            for j in pontos:\n",
    "                if j != atual and value(x[atual][j][k]) and value(x[atual][j][k]) > 0.5:\n",
    "                    proximo = j\n",
    "                    break\n",
    "            if proximo is None or proximo == 0:\n",
    "                rota.append(0)\n",
    "                break\n",
    "            rota.append(proximo)\n",
    "            atual = proximo\n",
    "        rotas[k] = rota\n",
    "    return rotas\n",
    "\n",
    "rotas = extrair_rotas(x, pontos, clientes, veiculos)\n",
    "\n",
    "for k in veiculos:\n",
    "    carga = sum(demandas[i] for i in rotas[k] if i != 0)\n",
    "    rota_nomes = ' → '.join(nomes[i] for i in rotas[k])\n",
    "    dist_rota = sum(dist[rotas[k][i]][rotas[k][i+1]] for i in range(len(rotas[k])-1))\n",
    "    print(f'Veículo {k+1}: {rota_nomes}')\n",
    "    print(f'  Carga: {carga}/{CAPACIDADE} kg | Distância: {round(dist_rota, 2)} km\\n')"
