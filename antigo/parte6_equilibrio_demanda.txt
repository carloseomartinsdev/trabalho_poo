    "# Função para gerar demandas proporcionais a um alvo\n",
    "def gerar_demandas_alvo(demandas_base, alvo_total):\n",
    "    \"\"\"Escala as demandas base proporcionalmente para atingir o alvo total.\"\"\"\n",
    "    base_sem_cd = demandas_base[1:]  # remove CD (demanda 0)\n",
    "    soma_base = sum(base_sem_cd)\n",
    "    if soma_base == 0:\n",
    "        return demandas_base\n",
    "    fator = alvo_total / soma_base\n",
    "    novas = [max(20, int(d * fator)) for d in base_sem_cd]\n",
    "    # Ajuste fino para bater o alvo\n",
    "    diff = alvo_total - sum(novas)\n",
    "    if diff > 0:\n",
    "        for i in range(diff):\n",
    "            idx = i % len(novas)\n",
    "            novas[idx] += 1\n",
    "    elif diff < 0:\n",
    "        for i in range(-diff):\n",
    "            idx_max = novas.index(max(novas))\n",
    "            novas[idx_max] -= 1\n",
    "    return [0] + novas\n",
    "\n",
    "\n",
    "def resolver_e_custo_mensal(n_veiculos, demandas_teste, cap, matriz_dist, coords_gps, nomes_pts):\n",
    "    \"\"\"Resolve o CVRP e retorna o custo mensal total (km + salários). Retorna None se inviável.\"\"\"\n",
    "    n = len(coords_gps)\n",
    "    cli = list(range(1, n))\n",
    "    pts = list(range(n))\n",
    "    veics = list(range(n_veiculos))\n",
    "\n",
    "    m = LpProblem(f'CVRP_{n_veiculos}v_{sum(demandas_teste)}kg', LpMinimize)\n",
    "    xv = LpVariable.dicts('x', (pts, pts, veics), cat='Binary')\n",
    "    uv = LpVariable.dicts('u', (cli, veics), lowBound=1, upBound=len(cli), cat='Continuous')\n",
    "\n",
    "    m += lpSum(matriz_dist[i][j] * xv[i][j][k] for i in pts for j in pts for k in veics if i != j)\n",
    "\n",
    "    for j in cli:\n",
    "        m += lpSum(xv[i][j][k] for i in pts for k in veics if i != j) == 1\n",
    "    for k in veics:\n",
    "        m += lpSum(xv[0][j][k] for j in cli) <= 1\n",
    "        m += lpSum(xv[j][0][k] for j in cli) <= 1\n",
    "    for k in veics:\n",
    "        for j in cli:\n",
    "            m += lpSum(xv[i][j][k] for i in pts if i != j) == lpSum(xv[j][i][k] for i in pts if i != j)\n",
    "    for k in veics:\n",
    "        m += lpSum(demandas_teste[j] * lpSum(xv[i][j][k] for i in pts if i != j) for j in cli) <= cap\n",
    "    for k in veics:\n",
    "        for i in cli:\n",
    "            for j in cli:\n",
    "                if i != j:\n",
    "                    m += uv[i][k] - uv[j][k] + len(cli) * xv[i][j][k] <= len(cli) - 1\n",
    "\n",
    "    m.solve(PULP_CBC_CMD(msg=0))\n",
    "    \n",
    "    if LpStatus[m.status] != 'Optimal':\n",
    "        return None, None  # Inviável\n",
    "    \n",
    "    dist_total = round(value(m.objective), 2)\n",
    "    custo_km_mes = round(dist_total * CUSTO_POR_KM * DIAS_UTEIS_MES, 2)\n",
    "    custo_salarios = n_veiculos * SALARIO_MOTORISTA\n",
    "    custo_total_mes = round(custo_km_mes + custo_salarios, 2)\n",
    "    \n",
    "    return dist_total, custo_total_mes"
   ]
