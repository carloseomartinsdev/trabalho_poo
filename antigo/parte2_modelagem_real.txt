    "cores = ['green', 'purple', 'orange', 'cyan', 'magenta']\n",
    "\n",
    "def extrair_rotas(x, pontos, clientes, veiculos):\n",
    "    rotas = {}\n",
    "    for k in veiculos:\n",
    "        rota = [0]\n",
    "        atual = 0\n",
    "        while True:\n",
    "            proximo = None\n",
    "            for j in pontos:\n",
    "                if j != atual and value(x[atual][j][k]) and value(x[atual][j][k]) > 0.5:\n",
    "                    proximo = j\n",
    "                    break\n",
    "            if proximo is None or proximo == 0:\n",
    "                rota.append(0)\n",
    "                break\n",
    "            rota.append(proximo)\n",
    "            atual = proximo\n",
    "        rotas[k] = rota\n",
    "    return rotas\n",
    "\n",
    "rotas = extrair_rotas(x, pontos, clientes, veiculos)\n",
    "\n",
    "for k in veiculos:\n",
    "    carga = sum(demandas[i] for i in rotas[k] if i != 0)\n",

    "    print(f'Veículo {k+1}: {rota_nomes}')\n",
